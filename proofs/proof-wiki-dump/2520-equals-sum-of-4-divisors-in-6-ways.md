theorem Sum_Four_Divisors_2520() {
  assert(
    ∃n = 2520 ∧
    ∃!k = 6 ∧
    ∀i ∈ [1..k] ∃(a_i, b_i, c_i, d_i) ∈ div(n) ⇒
    n = a_i + b_i + c_i + d_i ∧
    isMaximal(k)
  )
}

proof Sum_Four_Divisors_2520() {
  lemma Unit_Fractions() {
    assert(
      1 = 1/2 + 1/3 + 1/7 + 1/42 ∧
      1 = 1/2 + 1/3 + 1/8 + 1/24 ∧
      1 = 1/2 + 1/3 + 1/9 + 1/18 ∧
      1 = 1/2 + 1/3 + 1/10 + 1/15 ∧
      1 = 1/2 + 1/4 + 1/5 + 1/20 ∧
      1 = 1/2 + 1/4 + 1/6 + 1/12
    )
  } →

  lemma Prime_Factorizations() {
    assert(
      42 = 2¹ × 3¹ × 7¹ ∧
      24 = 2³ × 3¹ ∧
      18 = 2¹ × 3² ∧
      30 = 2¹ × 3¹ × 5¹ ∧
      20 = 2² × 5¹ ∧
      12 = 2² × 3¹
    )
  } →

  setVar(maxPowers = {
    2: max(1,3,1,1,2,2) = 3,
    3: max(1,1,2,1,0,1) = 2,
    5: max(0,0,0,1,1,0) = 1,
    7: max(1,0,0,0,0,0) = 1
  }) →

  assert(n = 2³ × 3² × 5¹ × 7¹ = 2520) →

  verify(
    2520 = 1260 + 630 + 504 + 126 ∧
    2520 = 1260 + 630 + 421 + 210 ∧
    2520 = 1260 + 840 + 360 + 60 ∧
    2520 = 1260 + 840 + 315 + 105 ∧
    2520 = 1260 + 840 + 280 + 140 ∧
    2520 = 1260 + 840 + 252 + 168
  )
}
