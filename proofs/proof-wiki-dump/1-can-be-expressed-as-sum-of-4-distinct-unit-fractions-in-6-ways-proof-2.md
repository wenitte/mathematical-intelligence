theorem Unit_Fractions_Sum() {
  assert(
    ∃!S: Set[Equation] where |S| = 6 ∧
    (∀eq ∈ S)(
      eq has form: 1 = Σᵢ₌₁⁴(1/nᵢ) where 
      nᵢ ∈ ℕ⁺ ∧
      ∀i,j(i ≠ j → nᵢ ≠ nⱼ)
    )
  )
}

proof Unit_Fractions_Sum() {
  lemma All_Solutions() {
    assert(
      ∃T: Set[Equation] where |T| = 14 ∧
      T = {
        1 = 1/2 + 1/3 + 1/7 + 1/42,
        1 = 1/2 + 1/3 + 1/8 + 1/24,
        1 = 1/2 + 1/3 + 1/9 + 1/18,
        1 = 1/2 + 1/3 + 1/10 + 1/15,
        1 = 1/2 + 1/3 + 1/12 + 1/12,
        1 = 1/2 + 1/4 + 1/5 + 1/20,
        1 = 1/2 + 1/4 + 1/6 + 1/12,
        1 = 1/2 + 1/4 + 1/8 + 1/8,
        1 = 1/2 + 1/5 + 1/5 + 1/10,
        1 = 1/2 + 1/6 + 1/6 + 1/6,
        1 = 1/3 + 1/3 + 1/4 + 1/12,
        1 = 1/3 + 1/3 + 1/6 + 1/6,
        1 = 1/3 + 1/4 + 1/4 + 1/6,
        1 = 1/4 + 1/4 + 1/4 + 1/4
      }
    )
  } →
  
  apply(All_Solutions()) →
  
  assert(
    S = {eq ∈ T | hasDistinctDenominators(eq)} →
    |S| = 6 ∧
    S = {
      1 = 1/2 + 1/3 + 1/7 + 1/42,
      1 = 1/2 + 1/3 + 1/8 + 1/24,
      1 = 1/2 + 1/3 + 1/9 + 1/18,
      1 = 1/2 + 1/3 + 1/10 + 1/15,
      1 = 1/2 + 1/4 + 1/5 + 1/20,
      1 = 1/2 + 1/4 + 1/6 + 1/12
    }
  )
}
