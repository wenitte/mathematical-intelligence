theorem Binary_19() {
  assert(
    19 = 10011₂
  )
} ↔

proof Binary_19() {
  assert(19) →
  assert(19 = 18 + 1) →
  assert(18 + 1 = (2 × 9) + 1) →
  assert((2 × 9) + 1 = (2 × (8 + 1)) + 1) →
  assert((2 × (8 + 1)) + 1 = (2 × (2³ + 1)) + 1) →
  assert((2 × (2³ + 1)) + 1 = 2⁴ + 2 + 1) →
  assert(2⁴ + 2 + 1 = (1 × 2⁴) + (0 × 2³) + (0 × 2²) + (1 × 2¹) + (1 × 2⁰)) →
  assert((1 × 2⁴) + (0 × 2³) + (0 × 2²) + (1 × 2¹) + (1 × 2⁰) = 10011₂)
}
