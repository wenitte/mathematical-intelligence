quotient_definition(Q) ↔
    X,Y topological spaces
    p: X → Y surjective
    U ⊆ Y open
    ↔ p⁻¹(U) ⊆ X open
→

[universal_property(U) ↔
    f: X → Z continuous
    ∀x₁,x₂ ∈ X
    p(x₁) = p(x₂) → f(x₁) = f(x₂)
    → ∃!g: Y → Z continuous
    g ∘ p = f]
→

[inheritance(I) ↔
    X compact → Y compact
    X connected → Y connected
    p⁻¹(y) compact ∀y ∈ Y
    → p closed map]
