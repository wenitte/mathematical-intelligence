connected_definition(C) ↔
    X topological space
    X = U ∪ V, U,V open
    U ∩ V = ∅ ∧ U,V ≠ ∅
    → X not connected
→

[connected_continuous(F) ↔
    f: X → Y continuous
    X connected
    → f(X) connected]
→

[interval_connected(I) ↔
    [a,b] ⊆ ℝ
    a < b
    → [a,b] connected]
→

[component_maximal(M) ↔
    C ⊆ X connected
    C ⊆ D ⊆ X
    D connected
    → C = D]
